% ==============================================================================
% FORMATO DEL DOCUMENTO
% ==============================================================================

% ------------------------------------------------------------------------------
% Configuración de capítulos
% ------------------------------------------------------------------------------
\usepackage{titlesec}

% Formato de capítulos
\titleformat{\chapter}[display]
    {\normalfont\huge\bfseries\color{azulmetro}}
    {\chaptertitlename\ \thechapter}
    {20pt}
    {\Huge}

% Espaciado antes y después de capítulos
\titlespacing*{\chapter}{0pt}{-20pt}{40pt}

% Formato de secciones
\titleformat{\section}
    {\normalfont\Large\bfseries\color{azulmetro}}
    {\thesection}
    {1em}
    {}

% Formato de subsecciones
\titleformat{\subsection}
    {\normalfont\large\bfseries\color{grisoscuro}}
    {\thesubsection}
    {1em}
    {}

% ------------------------------------------------------------------------------
% Configuración de captions
% ------------------------------------------------------------------------------
\usepackage{caption}
\captionsetup{
    font=small,
    labelfont=bf,
    format=plain,
    justification=centering
}

% Configuración específica para figuras
\captionsetup[figure]{
    name=Figura,
    labelsep=period
}

% Configuración específica para tablas
\captionsetup[table]{
    name=Tabla,
    labelsep=period,
    position=top
}

% ------------------------------------------------------------------------------
% Configuración de encabezados (fancyhdr)
% ------------------------------------------------------------------------------
\setlength{\headheight}{15.05pt}    % Altura del encabezado (corrige advertencia)

% ------------------------------------------------------------------------------
% Configuración de párrafos
% ------------------------------------------------------------------------------
\setlength{\parindent}{1.25cm}      % Sangría de párrafos
\setlength{\parskip}{6pt}           % Espaciado entre párrafos

% ------------------------------------------------------------------------------
% Configuración de la tabla de contenidos
% ------------------------------------------------------------------------------
\setcounter{tocdepth}{3}            % Profundidad de la tabla de contenidos
\setcounter{secnumdepth}{3}         % Profundidad de numeración de secciones

% ------------------------------------------------------------------------------
% Definición de entornos personalizados
% ------------------------------------------------------------------------------

% Entorno para definiciones
\newtheoremstyle{definicion}
    {10pt}                          % Espacio antes
    {10pt}                          % Espacio después
    {\normalfont}                   % Fuente del cuerpo
    {}                              % Sangría
    {\bfseries\color{azulmetro}}    % Fuente del encabezado
    {:}                             % Puntuación después del encabezado
    {.5em}                          % Espacio después del encabezado
    {}                              % Especificación del encabezado

\theoremstyle{definicion}
\newtheorem{definicion}{Definición}[chapter]

% Entorno para ejemplos
\newtheorem{ejemplo}{Ejemplo}[chapter]

% Entorno para notas importantes
\newenvironment{notaimportante}
    {\begin{center}
    \begin{tabular}{|p{0.9\textwidth}|}
    \hline\\
    \textbf{Nota Importante:}\\\\}
    {\\\\
    \hline
    \end{tabular}
    \end{center}}

% ------------------------------------------------------------------------------
% Configuración de referencias cruzadas
% ------------------------------------------------------------------------------
\usepackage{cleveref}
\crefname{figure}{figura}{figuras}
\Crefname{figure}{Figura}{Figuras}
\crefname{table}{tabla}{tablas}
\Crefname{table}{Tabla}{Tablas}
\crefname{equation}{ecuación}{ecuaciones}
\Crefname{equation}{Ecuación}{Ecuaciones}
\crefname{chapter}{capítulo}{capítulos}
\Crefname{chapter}{Capítulo}{Capítulos}
\crefname{section}{sección}{secciones}
\Crefname{section}{Sección}{Secciones}
