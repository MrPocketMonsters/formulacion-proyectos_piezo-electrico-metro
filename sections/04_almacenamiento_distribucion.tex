% Sistema de almacenamiento y distribución

\subsection{Diseño del Sistema de Almacenamiento}

\subsubsection{Requisitos del Sistema}

\begin{itemize}
    \item Capacidad de almacenamiento: 100 kWh
    \item Potencia de descarga: 20 kW
    \item Ciclos de vida: > 5,000 ciclos
    \item Eficiencia de carga/descarga: > 90\%
\end{itemize}

\subsubsection{Configuración Propuesta}

\begin{itemize}
    \item \textbf{Almacenamiento primario:} Banco de supercapacitores (20 kWh)
        \begin{itemize}
            \item Para cargas/descargas rápidas
            \item Alta densidad de potencia
        \end{itemize}
    \item \textbf{Almacenamiento secundario:} Baterías ion-litio (80 kWh)
        \begin{itemize}
            \item Para almacenamiento de largo plazo
            \item Alta densidad de energía
        \end{itemize}
\end{itemize}

\subsection{Sistema de Distribución}

\subsubsection{Cargas Prioritarias}

Orden de prioridad para uso de la energía generada:

\begin{enumerate}
    \item Iluminación LED de señalización
    \item Pantallas de información al usuario
    \item Iluminación de emergencia
    \item Sistemas de comunicación
    \item Red general de la estación (excedente)
\end{enumerate}

\subsubsection{Esquema de Distribución}

\begin{itemize}
    \item Tablero principal de distribución
    \item Circuitos dedicados por tipo de carga
    \item Medidores de energía bidireccionales
    \item Sistema de conmutación automática
\end{itemize}

\subsection{Sistema de Monitoreo}

\begin{itemize}
    \item Dashboard en tiempo real de generación
    \item Registro histórico de producción
    \item Alertas de mantenimiento preventivo
    \item Integración con sistema SCADA de la estación
\end{itemize}
