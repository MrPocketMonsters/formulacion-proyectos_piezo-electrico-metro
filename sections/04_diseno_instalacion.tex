\subsection{Layout de Instalación}

\subsubsection{Configuración Tipo A: Zona de Torniquetes}

\begin{itemize}
    \item Área de instalación: 40 m² por estación
    \item Disposición: Cobertura total del área de paso
    \item Número de losetas: 160 unidades
\end{itemize}

\subsubsection{Configuración Tipo B: Pasillo Principal}

\begin{itemize}
    \item Área de instalación: 60 m² por estación
    \item Disposición: Franjas centrales de alto tráfico
    \item Número de losetas: 240 unidades
\end{itemize}

\subsection{Detalles Constructivos}

\subsubsection{Preparación del Sustrato}

\begin{enumerate}
    \item Evaluación de la superficie existente
    \item Nivelación y refuerzo estructural si es necesario
    \item Instalación de capa de aislamiento eléctrico
    \item Preparación de canalizaciones para cableado
\end{enumerate}

\subsubsection{Instalación de Losetas}

\begin{enumerate}
    \item Colocación de base de soporte
    \item Instalación de módulos piezoeléctricos
    \item Conexión eléctrica en serie/paralelo
    \item Instalación de capa protectora
    \item Sellado y acabado final
\end{enumerate}

\subsection{Especificaciones de Instalación}

\begin{itemize}
    \item Profundidad de instalación: 8-10 cm
    \item Resistencia al tráfico: Clase 34/43 (EN 685)
    \item Nivel de antideslizante: R10-R11 (DIN 51130)
    \item Protección IP: IP67 (resistencia al agua)
\end{itemize}
